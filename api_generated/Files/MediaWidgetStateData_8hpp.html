Detailed Description

Comprehensive serializable state data structure for MediaWidget.

This file defines the full state structure that MediaWidgetState serializes to JSON using reflect-cpp. It captures all persistent state needed for workspace save/restore:

  • Display options for all data types (lines, masks, points, tensors, intervals, media)
  • Viewport state (zoom, pan, canvas size)
  • Interaction preferences (line tools, mask brush, point selection)
  • Text overlays
  • Active tool modes

Design Principles

  1. Nested objects for clarity - Unlike DisplayOptions which use rfl::Flatten, this top-level structure uses nested objects for clear JSON organization
  2. Native enum serialization - Enums serialize as strings automatically
  3. No Qt types - All Qt types converted to std equivalents (QString → std::string)
  4. Transient state excluded - Hover positions, drag state, preview flags are NOT included

#ifndef MEDIA_WIDGET_STATE_DATA_HPP
#define MEDIA_WIDGET_STATE_DATA_HPP

#include "DisplayOptions/DisplayOptions.hpp"

#include "CorePlotting/Layout/CanvasCoordinateSystem.hpp"

#include <rfl.hpp>
#include <rfl/json.hpp>

#include <map>
#include <string>
#include <vector>

// ==================== Text Overlay Data ====================

enum class TextOverlayOrientation {
    Horizontal, 
    Vertical    
};

struct TextOverlayData {
    int id = -1;                                              
    std::string text;                                         
    TextOverlayOrientation orientation = TextOverlayOrientation::Horizontal;  
    float x_position = 0.5f;                                  
    float y_position = 0.5f;                                  
    std::string color = "#ffffff";                            
    int font_size = 12;                                       
    bool enabled = true;                                      
};

// ==================== Interaction Preferences ====================

struct LineInteractionPrefs {
    std::string smoothing_mode = "SimpleSmooth";  
    int polynomial_order = 3;                      
    bool edge_snapping_enabled = false;           
    int edge_threshold = 100;                      
    int edge_search_radius = 20;                   
    int eraser_radius = 10;                        
    float selection_threshold = 15.0f;            
};

struct MaskInteractionPrefs {
    int brush_size = 15;              
    bool hover_circle_visible = true; 
    bool allow_empty_mask = false;    
};

struct PointInteractionPrefs {
    float selection_threshold = 10.0f;  
};

// ==================== Viewport State ====================

struct ViewportState {
    double zoom = 1.0;        
    double pan_x = 0.0;       
    double pan_y = 0.0;       
    int canvas_width = 640;   
    int canvas_height = 480;  
};

// ==================== Tool Mode Enums ====================

enum class LineToolMode {
    None,          
    Add,           
    Erase,         
    Select,        
    DrawAllFrames  
};

enum class MaskToolMode {
    None,   
    Brush   
};

enum class PointToolMode {
    None,   
    Select  
};

// ==================== Main State Structure ====================

struct MediaWidgetStateData {
    // === Identity ===
    std::string instance_id;                        
    std::string display_name = "Media Viewer";      
    
    // === Primary Display ===
    std::string displayed_media_key;                
    
    // === Viewport (nested object in JSON) ===
    ViewportState viewport;                         
    
    // === Canvas Coordinate System Override ===
    bool canvas_coord_override_active{false};
    int canvas_coord_override_width{640};
    int canvas_coord_override_height{480};
    
    // === Per-Feature Display Options ===
    // Each key is a data key (e.g., "whisker_1"), value is the display options.
    // The 'is_visible' field in each options struct indicates if that feature is enabled.
    std::map<std::string, MediaDisplayOptions> media_options;               
    std::map<std::string, LineDisplayOptions> line_options;                 
    std::map<std::string, MaskDisplayOptions> mask_options;                 
    std::map<std::string, PointDisplayOptions> point_options;               
    std::map<std::string, DigitalIntervalDisplayOptions> interval_options;  
    std::map<std::string, TensorDisplayOptions> tensor_options;             
    
    // === Interaction Preferences (nested objects in JSON) ===
    LineInteractionPrefs line_prefs;   
    MaskInteractionPrefs mask_prefs;   
    PointInteractionPrefs point_prefs; 
    
    // === Text Overlays ===
    std::vector<TextOverlayData> text_overlays;  
    int next_overlay_id = 0;                     
    
    // === Active Tool State ===
    LineToolMode active_line_mode = LineToolMode::None;    
    MaskToolMode active_mask_mode = MaskToolMode::None;    
    PointToolMode active_point_mode = PointToolMode::None; 
};

#endif // MEDIA_WIDGET_STATE_DATA_HPP
