bool success = false;

Whether the pipeline completed successfully.

variable error_message

std::string error_message;

Human-readable error message (empty if success)

variable failed_stage

ClusteringStage failed_stage = ClusteringStage::ValidatingFeatures;

Stage at which the pipeline failed (only meaningful if !success)

variable fitting_observations

std::size_t fitting_observations = 0;

Number of observations used for fitting (after NaN dropping)

variable num_features

std::size_t num_features = 0;

Number of features used.

variable num_clusters

std::size_t num_clusters = 0;

Number of clusters discovered or requested.

variable rows_dropped_nan

std::size_t rows_dropped_nan = 0;

Number of rows dropped due to NaN/Inf in feature conversion.

variable noise_points

std::size_t noise_points = 0;

Number of noise points (only meaningful for DBSCAN)

Points not assigned to any cluster. 0 for non-density-based methods.

variable assignment_observations

std::size_t assignment_observations = 0;

Number of observations assigned to clusters.

variable cluster_sizes

std::vector< std::size_t > cluster_sizes;

Per-cluster observation counts (index = cluster id, value = count)

Size = num_clusters. Does not include noise points.

variable cluster_names

std::vector< std::string > cluster_names;

Cluster names corresponding to output groups.

Generated as “{output_prefix}{cluster_index}”.

variable interval_keys

std::vector< std::string > interval_keys;

DataManager keys of created DigitalIntervalSeries (one per cluster)

Empty if write_intervals was false.

variable probability_keys

std::vector< std::string > probability_keys;

DataManager keys of created AnalogTimeSeries (one per cluster)

Empty if write_probabilities was false or model doesn’t support them.

variable putative_group_ids

std::vector< GroupId > putative_group_ids;

GroupIds of created putative groups (one per cluster)

Empty if write_to_putative_groups was false.

variable deferred_output

std::optional< PredictionOutput > deferred_output;

Prediction data awaiting main-thread writing.

Populated when config.defer_dm_writes is true. The caller must call writePredictions() on the main thread with this data.

variable deferred_cluster_names

std::vector< std::string > deferred_cluster_names;

Cluster names for deferred writing (paired with deferred_output)

variable deferred_output_config

std::optional< PredictionWriterConfig > deferred_output_config;

Writer config for deferred writing (paired with deferred_output)

variable fitted_model

std::unique_ptr< MLModelOperation > fitted_model;

The fitted model (ownership transferred to caller)

Allows the caller to reuse the model for further cluster assignments or save it to disk. nullptr if fitting failed.
